# Create The North
narrative_event = {
	id = major.1
	title = EVTNAMEmajor.1
	desc = EVTDESCmajor.1
	picture = GFX_evt_rome_restored_byzantine
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTAmajor.1
		primary_title = {
			e_the_north = {
				grant_title = ROOT
				copy_title_laws = PREV
				copy_title_history = PREV
			}
		}
		give_nickname = nick_the_uniter
		hidden_tooltip = {
			any_playable_ruler = {
				limit = {
					NOT = { character = ROOT }
					ai = no
				}
				narrative_event = { id = major.2 }
			}
		}
	}	
}
