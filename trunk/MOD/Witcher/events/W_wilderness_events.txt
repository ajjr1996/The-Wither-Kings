################################################

################################################
# Hidden - Give tower to wilderness (on_death)
character_event = {
	id = wilderness.100
	
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		any_demesne_province = { culture = magic }
		NOT = { has_landed_title = d_thanedd }
		NOT = { has_landed_title = d_ban_ard_academy }
		NOT = { has_landed_title = d_nilfgaard_academy }
	}
	
	option = {
		name = OK
		d_wilderness = {
			holder_scope = {
				ROOT = {
					any_demesne_title = {
						limit = {
							tier = count
							location = { culture = magic }
						}
						gain_title = PREVPREV
					}
				}
				set_defacto_liege = THIS
			}
		}
	}
}
# Hidden - Give non-sorcerer held tower to wilderness 
character_event = {
	id = wilderness.101
	
	hide_window = yes
	
	trigger = {
		any_demesne_province = { culture = magic }
		NOT = { has_landed_title = d_thanedd }
		NOT = { has_landed_title = d_ban_ard_academy }
		NOT = { has_landed_title = d_nilfgaard_academy }
		NOT = { trait = sorcerer }
	}
	
	mean_time_to_happen = {
		days = 1 
	}
	
	option = {
		name = OK
		d_wilderness = {
			holder_scope = {
				ROOT = {
					any_demesne_title = {
						limit = {
							tier = count
							location = { culture = magic }
						}
						gain_title = PREVPREV
					}
				}
				set_defacto_liege = THIS
			}
		}
	}
}
# Hidden - Change accidently spread magic culture
character_event = {
	id = wilderness.102
	
	hide_window = yes
	
	trigger = {
		culture = magic
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = { # to elven culture
		name = OK
		trigger = {
			trait = elven
		}
		culture = aen_seidhe
		ai_chance = { factor = 100 }
	}
	option = { # to dwarven culture
		name = OK
		trigger = {
			trait = dwarven
		}
		culture = dwarven
		ai_chance = { factor = 100 }
	}
	option = { # to liege's culture
		name = ""
		trigger = {
			liege = { NOT = { culture = magic } }
		}
		liege = { reverse_culture = ROOT }
		ai_chance = { factor = 10 }
	}
	option = { # to neighbor's culture
		name = OK
		location = {
			random_neighbor_province = { 
				limit = {
					NOT = { culture = magic }
					NOT = { culture = aen_seidhe }
					NOT = { culture = dwarven }
				}
				reverse_culture = ROOT
			}
		}
		ai_chance = { factor = 1 }
	}
}
