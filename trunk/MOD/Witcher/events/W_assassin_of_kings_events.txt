		name = EVTOPTAaok.6

			limit = { 
				OR = {

				any_child = {

				}
			}
		}
		ai_chance = { factor = 1 }
	}

		name = EVTOPTAaok.6

				OR = {

				}

						father = { character = ROOT }
					}

				}
			}

	}
