character_event = {
	id = 130000014
	desc = "ESMISC130000008DESC"
	picture = GFX_evt_comet
	hide_window = yes
	
	trigger = {
		NOT = { trait = human }
		NOT = { trait = dwarf }
		NOT = { trait = elven }
		NOT = { trait = halfelf }
		NOT = { trait = dryad }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		name = "My race is..."
		hidden_tooltip = {
			if = {
				limit = {
					graphical_culture = dryadgfx
				}
				add_trait = dryad
			}	
			if = {
				limit = {
					father_even_if_dead = { NOT = { graphical_culture = elvengfx } }
					mother = { graphical_culture = elvengfx }
					OR = {
						trait = human
						trait = elf
						NOT = { trait = halfelf }
					}
				}
				remove_trait = human
				remove_trait = elf
				add_trait = halfelf
			}
			if = {
				limit = {
					father_even_if_dead = { graphical_culture = elvengfx }
					mother = { NOT = { graphical_culture = elvengfx } }
					OR = {
						trait = human
						trait = elf
						NOT = { trait = halfelf }
					}
				}
				remove_trait = human
				remove_trait = elf
				add_trait = halfelf
			}
			if = {
				limit = {
					graphical_culture = elvengfx
					mother = { culture = elven }
					father = { culture = elven }
					NOT = { 
						mother = { trait = halfelf } 
					}
					NOT = { father = { trait = halfelf } }
				}
				add_trait = elven
			}
			if = {
				limit = {
					graphical_culture = dwarvengfx
				}
				add_trait = dwarven
			}
			if = {
				limit = {
					NOT = {
						OR = {
							graphical_culture = dryadgfx
							graphical_culture = dwarvengfx
							trait = elven
							trait = halfelf
							trait = dwarven
							trait = dryad
						}	
					}
				}
				add_trait = human
			}
		}
	}
}
