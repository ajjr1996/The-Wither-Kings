character_event = {
	id = 130000014
	desc = "ESMISC130000008DESC"
	picture = GFX_evt_comet
	hide_window = yes
	
	trigger = {
		NOT = { trait = human }
		NOT = { trait = dwarf }
		NOT = { trait = halfdwarf }
		NOT = { trait = elven }
		NOT = { trait = halfelf }
		NOT = { trait = dryad }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		name = "My race is..."
		hidden_tooltip = {
			if = {
				limit = {
					graphical_culture = dryadgfx
				}
				add_trait = dryad
			}	
			if = {
				limit = {
					OR = {
						OR = {
							AND = {
								father_even_if_dead = { NOT = { trait = elven } }
								mother_even_if_dead = { trait = elven }
							}
							AND = {
								father_even_if_dead = { trait = elven }
								mother_even_if_dead = { NOT = { trait = elven } }
							}
							AND = {
								father_even_if_dead = { trait = halfelf }
								mother_even_if_dead = { trait = halfelf }
							}
						}
						#OR = {
						#	AND = {
						#		trait = human
						#		trait = elf
						#	}
						#	NOT = { trait = halfelf }
						#}
					}
				}
				remove_trait = human
				remove_trait = elven
				remove_trait = elven_blood
				add_trait = halfelf
			}
			if = {
				limit = {
					graphical_culture = elvengfx
					OR = {
					
						#if both parents are elfs the child is an elf
						AND = {
							father_even_if_dead = { culture = elven }
							mother_even_if_dead = { culture = elven }
							NOT = { father_even_if_dead = { trait = halfelf } }
							NOT = { mother_even_if_dead = { trait = halfelf } }	
						}
						
						#Elf + halfelf "upgrades" to a full elf
						AND = {
							graphical_culture = elvengfx
							father_even_if_dead = { trait = elven }
							mother_even_if_dead = { trait = halfelf }
						}
						AND = {
							graphical_culture = elvengfx
							mother_even_if_dead = { trait = elven }
							father_even_if_dead = { trait = halfelf }
						}
					}
				}
				add_trait = elven
			}
			if = {
				limit = {
					OR = {
						AND = {
							father_even_if_dead = { trait = halfelf }
							mother_even_if_dead = { 
								NOT = { 
									OR = { 
										trait = halfelf 
										trait = elven 
									} 
								} 
							}
						}
						AND = {
							father_even_if_dead = { 
								NOT = { 
									OR = { 
										trait = halfelf 
										trait = elven 
									}
								} 
							}
							mother_even_if_dead = { trait = halfelf }
						}
						father_even_if_dead = { trait = elven_blood }
						mother = { trait = elven_blood }
					}
				}
				add_trait = elven_blood
			}
			if = {
				limit = {
					OR = {
						OR = {
							AND = {
								father_even_if_dead = { NOT = { trait = dwarven } }
								mother_even_if_dead = { trait = dwarven }
							}
							AND = {
								father_even_if_dead = { trait = dwarven }
								mother_even_if_dead = { NOT = { trait = dwarven } }
							}
							AND = {
								father_even_if_dead = { trait = halfdwarf }
								mother_even_if_dead = { trait = halfdwarf }
							}
						}
						#OR = {
						#	AND = {
						#		trait = human
						#		trait = dwarven
						#	}
						#	NOT = { trait = halfdwarf }
						#}
					}
				}
				remove_trait = human
				remove_trait = dwarven
				remove_trait = dwarven_blood
				add_trait = halfdwarf
			}
			if = {
				limit = {
					graphical_culture = dwarvengfx
					OR = {
					
						#if both parents are dwarfs the child is an dwarf
						AND = {
							father_even_if_dead = { culture = dwarven }
							mother_even_if_dead = { culture = dwarven }
							NOT = { father_even_if_dead = { trait = halfdwarf } }
							NOT = { mother_even_if_dead = { trait = halfdwarf } }	
						}
						
						#dwarf + halfdwarf "upgrades" to a full dwarf
						AND = {
							graphical_culture = dwarvengfx
							father_even_if_dead = { trait = dwarven }
							mother_even_if_dead = { trait = halfdwarf }
						}
						AND = {
							graphical_culture = dwarvengfx
							mother_even_if_dead = { trait = dwarven }
							father_even_if_dead = { trait = halfdwarf }
						}
					}
				}
				remove_trait = halfdwarf
				add_trait = dwarven
			}
			if = {
				limit = {
					NOT = {
						OR = {
							graphical_culture = dryadgfx
							graphical_culture = dwarvengfx
							trait = elven
							trait = halfelf
							trait = dwarven
							trait = dryad
						}	
					}
				}
				add_trait = human
			}
		}
	}
}
