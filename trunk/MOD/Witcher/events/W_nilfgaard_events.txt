# Emperor Evaluation
character_event = {
	id = nilfgaard.1
	desc = "Emperor evaluation"
	picture = GFX_evt_holy_emperor
	
	hide_window = yes
	
	trigger = {
		NOT = { culture = wilderness}
		has_landed_title = e_nilfgaard
		NOT = {
			trait = strong_emperor
			trait = weak_emperor
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "Strong emperor"
		ai_chance = {
			factor = 1
			modifier = { factor = 4 trait = ambitious }
			modifier = { factor = 3 trait = genius }
			modifier = { factor = 2 trait = quick }
			modifier = { factor = 3 trait = diligent }
			modifier = { factor = 5 religion = solar }
			modifier = { factor = 5 culture = nilfgaardian }
			modifier = { factor = 2 trait = elusive_shadow }
			modifier = { factor = 2 trait = grey_eminence }
			modifier = { factor = 2 trait = midas_touched }
			modifier = { factor = 2 trait = brilliant_strategist }
			modifier = { factor = 2 trait = the_sage }
			modifier = {
				factor = 2
				is_adult = yes
				NOT = { has_character_modifier = old_age }
			}
		}
		character_event = { id = nilfgaard.2 }
	}
	option = {
		name = "Weak emperor"
		ai_chance = {
			factor = 2
			modifier = { factor = 4 trait = content }
			modifier = { factor = 2 trait = slow }
			modifier = { factor = 3 trait = imbecile }
			modifier = { factor = 3 trait = inbred }
			modifier = { factor = 2 trait = lunatic }
			modifier = { factor = 3 trait = infirm }
			modifier = { factor = 4 trait = incapable }
			modifier = { factor = 5 is_female = yes }
			modifier = { factor = 5 NOT = { trait = human } }
			modifier = { factor = 4 is_adult = no }
			
		}
		character_event = { id = nilfgaard.3 }
	}
}

# Emperor trait lost
character_event = {
	id = nilfgaard.4
	desc = EVTDESCnilfgaard.4
	picture = GFX_evt_holy_emperor
	
	trigger = {
		OR = {
			trait = strong_emperor
			trait = weak_emperor
		}
		NOT = { has_landed_title = e_nilfgaard }
		is_alive = yes
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		any_independent_ruler = {
			any_realm_character = { clr_character_flag = previous_nilfgaard_emperor }
			clr_character_flag = previous_nilfgaard_emperor
		}
		set_character_flag = previous_nilfgaard_emperor
	}
	
	option = {
		name = EVTOPTAnilfgaard.4
		trigger = { trait = strong_emperor }
		remove_trait = strong_emperor
	}
	option = {
		name = EVTOPTAnilfgaard.4
		trigger = { trait = weak_emperor }
		remove_trait = weak_emperor
	}
}

			modifier = { factor = 2.0 e_nilfgaard = { holder_scope = { realm_size = 1000 } } }
		}
	}
	option = { # Call allies
		name = EVTOPTBnilfgaard.100
		any_independent_ruler = {
			limit = {
				culture_group = northern_group
				tier = KING

		}

			factor = 40

		}
