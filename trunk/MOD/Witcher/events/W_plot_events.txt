		months = 120

	}
	
	option = { # Yes
		name = EVTOPTAplots.100

		set_character_flag = discredit_fellow_vassal_attempt
	}
	option = { # No
		name = EVTOPTBplots.100
		ai_chance = {
			factor = 0
		}
	}
}
