# A dwarf warrior
character_event = {
	id = tolerant.1
	desc = EVTDESCtolerant.1
	picture = GFX_evt_emissary
	border = GFX_event_normal_frame_religion
	
	min_age = 15
	prisoner = no
	only_playable = yes
	capable_only = yes
	
	trigger = {
		trait = tolerant
		NOT = { trait = dryad }
	}
	
	mean_time_to_happen = {
		months = 1100
		modifier = { factor = 2 any_courtier = { trait = dwarven } }
		modifier = { factor = 2 any_courtier = { trait = elven } }
		modifier = { factor = 0.8 crownlaw_title = { has_law = tolerance_0 } }
		modifier = { factor = 0.9 trait = halfdwarf }
		modifier = { factor = 0.9 trait = gregarious }
		modifier = { factor = 0.9 trait = kind }
		modifier = { factor = 0.9 trait = just }
	}
	
	option = {
		name = EVTOPTAtolerant.1
		create_character = {
			random_traits = yes
			culture = dwarven
			dynasty = none
			religion = dwarven
			female = no
			age = 30
			trait = dwarven
			trait = skilled_tactician
			trait = skilled_warrior
			attributes = { martial = 10 }
		}
	}
}
