			}

################################################
# Court mage casts meteor spell
character_event = {
	id = spell.200
	desc = EVTDESCspell.200
	picture = GFX_evt_battle
	border = GFX_event_normal_frame_war
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTAspell.200
		job_mage = {
			add_character_modifier = { name = drained duration = 360 }
			hidden_tooltip = { character_event = { id = spell.202 } } # Court mage notification
			ROOT = {
				if = {
					limit = {
						is_ruler = yes
						war = yes
						OR = {
							is_primary_war_attacker = yes
							is_primary_war_defender = yes
							independent = yes
						}
					}
					any_current_enemy = {
						limit = {
							OR = {
								at_location = PREV
								AND = {
									ROOT = { ai = yes }
									at_location = ROOT
								}
							}
							in_command = yes
							prisoner = no
						}
						morale = -0.4
						troops = -0.4
						character_event = { id = spell.203 }
					}
					any_current_enemy = {
						any_realm_character = {
							limit = {
								OR = {
									at_location = PREVPREV
									AND = {
										ROOT = { ai = yes }
										at_location = ROOT
									}
								}
								in_command = yes
								prisoner = no
								NOT = { character = PREV }
							}
							morale = -0.4
							troops = -0.4
							character_event = { id = spell.203 }
						}
					}
				}
				if = {
					limit = {
