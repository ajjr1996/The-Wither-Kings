province_event = {
	id = rebels.1
	desc = "Why the hell ..."
	picture = GFX_evt_peasants
	
	is_triggered_only = yes

		owner = {
			NOT = { culture = wilderness}

		if = { # Peasant revolt
			limit = {
				owner = {
					top_liege = {
						war = yes

							defender = { character = PREV }
							AND = {
								using_cb = peasant_revolt
								war_title = ROOT
							}
						}
					}
				}
			}
			owner = { set_character_flag = ongoing_peasant_revolt }
		}
		if = {# Religious revolt
			limit = {
				owner = {
					top_liege = {
						war = yes
						any_war = {
							defender = { character = PREV }
							AND = { 
								using_cb = religious_revolt
								attacker = { religion = ROOT }
							}
						}
					}
				}
			}
			owner = { set_character_flag = ongoing_religious_revolt }
		}
		if = { # Nationalist revolt
			limit = {
				owner = {
					top_liege = {
						war = yes
						any_war = {
							defender = { character = PREV }
							AND = {
								using_cb = liberation_revolt
								thirdparty_title_scope = {
									ROOT = {
										kingdom = {
											title = PREVPREV
										}
									}
								}
							}
						}
					}
				}
			}
			owner = { set_character_flag = ongoing_nationalist_revolt }
		}
		if = { # Scoitael revolt
			limit = {
				owner = {
					top_liege = {
						war = yes
						any_war = {
							defender = { character = PREV }
							AND = {
								using_cb = scoiatael_revolt

							}
						}
					}
				}
			}
			owner = { set_character_flag = ongoing_scoiatael_revolt }
		}
	}
	
	# Start a new revolt
	option = {
		name = "Start scoiatael revolt"
		trigger = {
			owner = {
				NOT = {
					has_character_flag = ongoing_peasant_revolt
					has_character_flag = ongoing_religious_revolt
					has_character_flag = ongoing_nationalist_revolt
					has_character_flag = ongoing_scoiatael_revolt
				}
				top_liege = {
					trait = human # Only against human rulers
					NOT = {
						trait = kind2 # Not against Saskia
						trait = tolerant
					}
				}
			}
		}
		ai_chance = {
			factor = 1000
			modifier = {
				factor = 0.001
				NOT = { # If not non-human religion or culture, then make as a bit less common than normal peasant revolt
					culture = elven
					culture = dwarven
					religion = dana_meadbh
					religion = dwarven

		}
