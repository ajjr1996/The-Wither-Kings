################################################
# Main rebel selection event (on_rebel_revolt)
################################################
province_event = {
	id = rebels.1
	desc = "Why the hell ..."
	picture = GFX_evt_peasants
	
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		owner = { NOT = { culture = wilderness} }
		has_dlc = "The Old Gods"
	}
	
	immediate = { # Check for any ongoing revolt
		owner = {
			top_liege = {
				if = { # Peasant revolt
					limit = {
						war = yes
						any_war = {
							defender = { character = PREV }
							AND = {
								using_cb = peasant_revolt
								war_title = ROOT
							}
						}
					}
					set_character_flag = ongoing_peasant_revolt
				}
				if = {# Religious revolt
					limit = {
						war = yes
						any_war = {
							defender = { character = PREV }
							AND = { 
								using_cb = religious_revolt
								attacker = { religion = ROOT }
							}
						}
					}
					set_character_flag = ongoing_religious_revolt
				}
				if = { # Nationalist revolt
					limit = {
						war = yes
						any_war = {
							defender = { character = PREV }
							AND = {
								using_cb = liberation_revolt
								thirdparty_title_scope = {
									ROOT = {
										kingdom = {
											title = PREVPREV
										}
									}
								}
							}
						}
					}
					set_character_flag = ongoing_nationalist_revolt
				}
				if = { # Scoitael revolt
					limit = {
						war = yes
						any_war = {
							defender = { character = PREV }
							AND = {
								using_cb = scoiatael_revolt
								war_title = ROOT
							}
						}
					}
					set_character_flag = ongoing_scoiatael_revolt
				}
				if = {
					limit = { had_character_flag = { flag = revolt_cooldown days = 1800 } }
					clr_character_flag = revolt_cooldown
				}
			}
		}
	}
	
	# Start a new revolt
	option = {
		name = "Start scoiatael revolt"
		trigger = {
			owner = {
				top_liege = {
					NOT = {
						has_character_flag = ongoing_peasant_revolt
						has_character_flag = ongoing_religious_revolt
						has_character_flag = ongoing_nationalist_revolt
						has_character_flag = ongoing_scoiatael_revolt
					}
					trait = human # Only against human rulers
					NOT = {
						trait = kind2 # Not against Saskia
						trait = tolerant
					}
				}
			}
		}
		ai_chance = {
			factor = 10000
			modifier = {
				factor = 0.0005
				NOT = { # If not non-human religion or culture, then make as a bit less common than normal peasant revolt
					culture = elven
					culture = dwarven
					religion = dana_meadbh
					religion = dwarven
				}
			}
		}
		province_event = { id = rebels.10 }
		owner = { top_liege = { set_character_flag = revolt_cooldown } }
	}
	option = {
		name = "Start nationalist revolt"
		trigger = {
			owner = {
				top_liege = {
					NOT = {
						has_character_flag = ongoing_peasant_revolt
						has_character_flag = ongoing_religious_revolt
						has_character_flag = ongoing_nationalist_revolt
						has_character_flag = ongoing_scoiatael_revolt
					}
					NOT = { culture = ROOT }
				}
			}
			kingdom = {
				culture = ROOT
				has_holder = no
				any_previous_holder = { always = yes } # Must have had a holder in the past
			}
		}
		ai_chance = {
			factor = 1000
		}
		province_event = { id = TOG.1040 }
		owner = { top_liege = {set_character_flag = revolt_cooldown } }
	}
	option = {
		name = "Start religious revolt"
		trigger = {
			owner = {
				top_liege = {
					NOT = {
						has_character_flag = ongoing_peasant_revolt
						has_character_flag = ongoing_religious_revolt
						has_character_flag = ongoing_nationalist_revolt
						has_character_flag = ongoing_scoiatael_revolt
					}
					NOT = { religion_group = ROOT }
				}
			}
			NOT = { # These two religions should rather use Scoiatael revolt
				religion = dana_meadbh
				religion = dwarven
			}
		}
		ai_chance = {
			factor = 100
			modifier = { factor = 0.1 religion = spider } # It's not really religion that would openly try to revolt 
		}
		province_event = { id = TOG.1060 }
		owner = { top_liege = {set_character_flag = revolt_cooldown } }
	}
	option = {
		name = "Start peasant revolt"
		trigger = {
			owner = {
				top_liege = {
					NOT = {
						has_character_flag = ongoing_peasant_revolt
						has_character_flag = ongoing_religious_revolt
						has_character_flag = ongoing_nationalist_revolt
						has_character_flag = ongoing_scoiatael_revolt
					}
					NOT = { # Elves don't get normal peasant revolt in elven culture and religion provinces
						ROOT = {
							culture = elven
							religion = dana_meadbh
						}
						trait = elven
					}
					NOT = { # Dwarves don't get normal peasant revolt in dwarven culture and religion provinces
						ROOT = {
							culture = dwarven
							religion = dwarven
						}
						trait = dwarven
					}
				}
			}
		}
		ai_chance = {
			factor = 10
		}
		province_event = { id = TOG.1000 }
		owner = { top_liege = {set_character_flag = revolt_cooldown } }
	}
	
	 # Reinforce the ongoing revolt if any
	option = {
		name = "Reinforce scoiatael revolt"
		trigger = {
			owner = {
				top_liege = {
					has_character_flag = ongoing_scoiatael_revolt
				}
			}
		}
		ai_chance = {
			factor = 10000
		}
		province_event = { id =  rebels.13 }
	}
	option = {
		name = "Reinforce nationalist revolt"
		trigger = {
			owner = {
				top_liege = {
					has_character_flag = ongoing_nationalist_revolt
				}
			}
		}
		ai_chance = {
			factor = 1000
		}
		province_event = { id = TOG.1050 }
	}
	option = {
		name = "Reinforce religious revolt"
		trigger = {
			owner = {
				top_liege = {
					has_character_flag = ongoing_religious_revolt
				}
			}
		}
		ai_chance = {
			factor = 100
		}
		province_event = { id = TOG.1070 }
	}
	option = {
		name = "Reinforce peasant revolt"
		trigger = {
			owner = {
				top_liege = {
					has_character_flag = ongoing_peasant_revolt
				}
			}
		}
		ai_chance = {
			factor = 10
		}
		province_event = { id = TOG.1010 }
	}
	
	 # Do nothing
	option = {
		name = "Do nothing"
		ai_chance = {
			factor = 1
			modifier = {
				factor = 100000
				owner = { top_liege = { has_character_flag = revolt_cooldown } }
			}
		}
	}
}
