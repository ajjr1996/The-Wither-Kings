#Tribal Raid CB
tribal_raid = {
	name = CB_NAME_TRIBAL_RAID
	war_name = WAR_NAME_TRIBAL_RAID
	sprite = 8
	truce_days = 1825
	hostile_against_others = yes
	is_permanent = yes
	can_ask_to_join_war = no
	can_call_allies = no
	check_de_jure_tier = DUKE # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	
	can_use = {
		always = no #Commented out for now
		ROOT = {
			OR = {
				culture_group = altaic
				culture_group = west_african
				culture_group = east_african
			}
			OR = {
				religion_group = pagan_group
				religion = monophysite
				AND = {
					religion_group = muslim
					OR = {
						culture = mongol
						culture_group = west_african
						culture_group = east_african
					}
				}
			}
			war = no
			independent = yes
			NOT = { has_truce = FROM }
			any_realm_province = {
				any_neighbor_province = {
					owner = {
						top_liege = {
							character = FROM
						}
					}
				}
			}
		}
		FROM = {
			NOT = { culture_group = altaic }
			OR = {
				NOT = { religion_group = pagan_group }
				religion = west_african_pagan
			}
			wealth = 30
			scaled_wealth = 1
		}
	}
	
	can_use_title = {
		any_direct_de_jure_vassal_title = {
			holder_scope = {
				OR = {
					character = FROM
					is_liege_or_above = FROM
				}
			}
			location = {
				any_neighbor_province = {
					owner = {
						OR = {
							character = ROOT
							is_liege_or_above = ROOT
						}
					}
				}
			}
		}
	}
	
	on_success = {
		ROOT = {
			prestige = 100
			piety = 50
			wealth = 30
		}
		FROM = {
			transfer_scaled_wealth = {
				to = ROOT
				value = 2.0
			}
			wealth = -30
			prestige = -50
			piety = -15
		}
	}
	
	on_success_title = {
		any_direct_de_jure_vassal_title = {
			limit = {
				holder_scope = {
					OR = {
						character = FROM
						is_liege_or_above = FROM
					}
				}
			}
			location = {
				add_province_modifier = {
					name = war_torn
					duration = 1825
				}
			}
		}
	}
	
	on_fail = {
		ROOT = {
			prestige = -25
			piety = -5
		}
	}
	
	on_reverse_demand = {
		ROOT = {
			prestige = -50
			piety = -15
			wealth = -30
			transfer_scaled_wealth = {
				to = FROM
				value = 2.0
			}
		}
		FROM = {
			wealth = 30
			prestige = 50
			piety = 15
		}
	}
	
	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}

	sprite = 6

	can_call_allies = no
	can_ask_to_join_war = no

	can_use_title = {
		holder_scope = { independent = yes }
		tier = king
		NOT = { has_law = king_peace_0 }

	on_success = {
		ROOT = {
			opinion = {
				modifier = opinion_lowered_crown_authority

	truce_days = 1850
	hostile_against_others = yes
	is_permanent = yes
	can_ask_to_join_war = yes

	can_use = {
		ROOT = {
			has_character_flag = bulgarian_uprising
		}
	}
	
	on_success_title = {
		gain_title = ROOT
		hidden_tooltip = {
			any_de_jure_vassal_title = {
				limit = {
					NOT = { tier = baron }
					holder_scope = { NOT = { culture = bulgarian } }
					holder_scope = { NOT = { any_liege = { culture = bulgarian } } }
					holder_scope = { top_liege = { has_landed_title = e_byzantium } }
				}
				usurp_title = ROOT
			}
			any_de_jure_vassal_title = {
				limit = {
					NOT = { tier = baron }
					holder_scope = { NOT = { any_liege = { culture = bulgarian } } }
					holder_scope = { top_liege = { has_landed_title = e_byzantium } }
					holder_scope = { NOT = { character = ROOT } }
				}
				holder_scope = { set_defacto_liege = ROOT }
			}
			add_law = crown_levies_1
			add_law = emperor_peace_0
			add_law = inheritance_0
			add_law = revokation_0
		}
	}
	
	on_success = {
		ROOT = {
			prestige = 200
		}
		FROM = {
			prestige = -200
		}
	}
	
	on_fail = {
		k_bulgaria = { destroy_landed_title = THIS }
		ROOT = {
			prestige = -50
			set_defacto_liege = FROM
		}
		FROM = {
			prestige = 100
		}
	}
	
	on_reverse_demand = {
		k_bulgaria = { destroy_landed_title = THIS }
		ROOT = {
			prestige = -100
			set_defacto_liege = FROM
		}
		FROM = {
			prestige = 200
		}
		set_global_flag = bulgarian_unrest_ends
	}
	
	on_add = {
		ROOT = {
			clr_character_flag = bulgarian_uprising
		}
	}
	
	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
