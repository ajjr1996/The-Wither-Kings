# Become Sorcerer
obj_become_sorcerer = {
	type = character
	
	potential = {
		NOT = { culture = wilderness }
	}

	allow = {
		is_adult = yes
		prisoner = no
		ai = no
		OR = {
			trait = uneducated_buffoon
			trait = jack_of_all_trades
			trait = well_versed_polymath
			trait = wise_sage
		}
		NOT = { trait = sorcerer }
		NOT = { trait = incapable }
	}
	
	chance = {
		factor = 100
		modifier = {
			factor = 4
			trait = ambitious
		}
		modifier = {
			factor = 3
			learning = 16
		}
		modifier = {
			factor = 2
			learning = 14
		}
		modifier = {
			factor = 1.5
			learning = 12
		}
		modifier = {
			factor = 0.5
			trait = slothful
		}
	}
	
	success = {
		trait = sorcerer
	}

	abort = {
		OR = {
			prisoner = yes
			trait = incapable
			trait = blinded
		}
	}
		
	effect = {
		prestige = 100
	}
}	
plot_kidnap_child = {
	type = realm_characters_plus 
	intrigue_plot = yes
	murder_plot = yes
	
	potential = {
		is_ruler = yes
		prisoner = no
		age = 16
		has_character_flag = chose_child_option	
		NOT = { trait = incapable }	
		NOT = { ai = yes }
	}
			

	# Target allow trigger for when players target a specific character in the GUI
	player_allow = {
		prisoner = no
		OR = {
			trait = magic_talent
			trait = adept
			trait = source
		} 
	}
	
	# Target scope
	allow = {
		prisoner = no
		max_age = 10
		OR = {
			trait = magic_talent
			trait = adept
			trait = source
		} 
	}
	
	chance = {
		modifier = {
			factor = 2
			FROM = { trait = ruthless }
		}
		modifier = {
			factor = 1.5
			FROM = { trait = arbitrary }
		}
		modifier = {
			factor = 2.0
			FROM = { trait = cruel }
		}
		modifier = {
			factor = 2.0
			FROM = { trait = deceitful }
		}
		modifier = {
			factor = 3.0
			FROM = { trait = ambitious }
		}		
		modifier = {
			factor = 0.5
			FROM = { NOT = { intrigue = 5 } }
		}		
		modifier = {
			factor = 0.5
			FROM = { NOT = { intrigue = 10 } }
		}
		modifier = {
			factor = 2
			FROM = { intrigue = 15 }
		}
		modifier = {
			factor = 2
			FROM = { intrigue = 20 }
		}		
	}
	
	success = {
		prisoner = yes
		host = { character = FROM }
	}
	abort = {
		OR = {	
			AND = {
				prisoner = yes
				host = { NOT = { character = FROM } }
			}
			FROM = { has_character_flag = abort_child_plot }
			FROM = { prisoner = yes }
			is_alive = no
			FROM = { trait = incapable }
		}
	}
	effect = {
		FROM = {
			any_plot_backer = {
				reverse_opinion = {
					modifier = opinion_grateful
					who = FROM
					years = 5
				}
				reverse_opinion = { who = ROOT modifier = opinion_recent_plot months = 60 }
			}
		}
	}
}
